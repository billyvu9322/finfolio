**Goal:** Implement the Gold module — transaction CRUD with a weighted-average-cost (DCA) + unrealized-P&L engine, a pluggable gold-price source with a 15-minute scheduled refresh, and the `/gold` portfolio + `/gold/add` web screens.

**Architecture:** Pure `goldMath` engine (decimal.js) does all unit conversion + DCA (FIFO) + P&L; `gold.service` wraps it with Drizzle reads/writes and the sell-guard; `gold.routes` exposes Zod-typed endpoints. A `PriceProvider` interface with a manual seed impl feeds `price_cache` via a node-cron job (env-gated) and a manual refresh endpoint. Web pages use a typed `gold.api` client + TanStack Query.

**Tech Stack:** Fastify 5, Drizzle (postgres-js), Zod, decimal.js, node-cron, vitest (API). React 18, TanStack Router/Query, Zustand, axios (Web). Runtime via `tsx`.

**Spec:** [../specs/2026-06-08-phase2-gold-design.md](../specs/2026-06-08-phase2-gold-design.md)

**Reference facts:**
- `gold_transactions` columns (Drizzle, strings for numeric): `id, userId, goldType, action('buy'|'sell'), quantity, unit('chi'|'luong'|'cay'), pricePerUnit, fee, storage, note, transactionAt, createdAt`.
- `price_cache` columns: `id, assetType('gold'|'stock'|'crypto'), symbol, priceBuy, priceSell, currency, source, fetchedAt`; unique index `(assetType, symbol)`.
- Unit ruling: `CHI_PER_UNIT = { chi: 1, luong: 10, cay: 10 }`; prices normalize to VND/Chỉ.

```ts
import Decimal from 'decimal.js';

export type GoldUnit = 'chi' | 'luong' | 'cay';

export const CHI_PER_UNIT: Record<GoldUnit, number> = { chi: 1, luong: 10, cay: 10 };

export interface GoldTx {
  action: 'buy' | 'sell';
  quantity: string | number;
  unit: GoldUnit;
  pricePerUnit: string | number;
  fee: string | number;
  transactionAt: Date;
}

/** Quantity expressed in Chỉ. */
export function toChi(qty: string | number, unit: GoldUnit): Decimal {
  return new Decimal(qty).mul(CHI_PER_UNIT[unit]);
}

/** Price normalized to VND per Chỉ. */
export function pricePerChi(price: string | number, unit: GoldUnit): Decimal {
  return new Decimal(price).div(CHI_PER_UNIT[unit]);
}

interface Lot {
  chi: Decimal;
  costPerChi: Decimal; // includes folded fee
}

export interface Holding {
  heldChi: Decimal;
  dcaPerChi: Decimal;
  investedRemaining: Decimal;
}

/**
 * Weighted-average cost with FIFO consumption on sells.
 * DCA of the remaining quantity is unaffected by sells; a partially consumed
 * buy lot keeps its per-Chỉ cost (fee already folded in).
 */
export function computeHolding(txs: GoldTx[]): Holding {
  const ordered = [...txs].sort((a, b) => a.transactionAt.getTime() - b.transactionAt.getTime());
  const lots: Lot[] = [];

  for (const t of ordered) {
    const chi = toChi(t.quantity, t.unit);
    if (t.action === 'buy') {
      const cost = pricePerChi(t.pricePerUnit, t.unit).mul(chi).plus(t.fee);
      lots.push({ chi, costPerChi: chi.isZero() ? new Decimal(0) : cost.div(chi) });
    } else {
      let remaining = chi;
      while (remaining.gt(0) && lots.length > 0) {
        const lot = lots[0]!;
        if (lot.chi.lte(remaining)) {
          remaining = remaining.minus(lot.chi);
          lots.shift();
        } else {
          lot.chi = lot.chi.minus(remaining);
          remaining = new Decimal(0);
        }
      }
    }
  }

  const heldChi = lots.reduce((s, l) => s.plus(l.chi), new Decimal(0));
  const investedRemaining = lots.reduce((s, l) => s.plus(l.chi.mul(l.costPerChi)), new Decimal(0));
  const dcaPerChi = heldChi.isZero() ? new Decimal(0) : investedRemaining.div(heldChi);
  return { heldChi, dcaPerChi, investedRemaining };
}

export function unrealizedPnl(
  heldChi: Decimal,
  dcaPerChi: Decimal,
  currentPricePerChi: Decimal,
): { pnl: Decimal; pnlPct: Decimal } {
  const pnl = currentPricePerChi.minus(dcaPerChi).mul(heldChi);
  const basis = dcaPerChi.mul(heldChi);
  const pnlPct = basis.isZero() ? new Decimal(0) : pnl.div(basis).mul(100);
  return { pnl, pnlPct };
}
```

```ts
import { db } from '../../../db/index.js';
import { priceCache } from '../../../db/schema/index.js';
import type { GoldPriceProvider } from './PriceProvider.js';

/** Fetches quotes from the provider and upserts them into price_cache. Returns count. */
export async function refreshGoldPrices(provider: GoldPriceProvider): Promise<number> {
  const quotes = await provider.fetchGoldPrices();
  const now = new Date();
  for (const q of quotes) {
    await db
      .insert(priceCache)
      .values({
        assetType: 'gold',
        symbol: q.symbol,
        priceBuy: q.priceBuy,
        priceSell: q.priceSell,
        currency: q.currency,
        source: q.source,
        fetchedAt: now,
      })
      .onConflictDoUpdate({
        target: [priceCache.assetType, priceCache.symbol],
        set: {
          priceBuy: q.priceBuy,
          priceSell: q.priceSell,
          currency: q.currency,
          source: q.source,
          fetchedAt: now,
        },
      });
  }
  return quotes.length;
}
```

```ts
/** Registers the 15-minute gold-price refresh when ENABLE_PRICE_SCHEDULER is on. */
export const schedulerPlugin = fp(async (fastify) => {
  if (!env.ENABLE_PRICE_SCHEDULER) {
    fastify.log.info('Price scheduler disabled');
    return;
  }
  const provider = new ManualGoldPriceProvider();
  const task: ScheduledTask = cron.schedule('*/15 * * * *', () => {
    refreshGoldPrices(provider)
      .then((n) => fastify.log.info(`Gold prices refreshed: ${n}`))
      .catch((err) => fastify.log.error(err, 'Gold price refresh failed'));
  });
  fastify.addHook('onClose', async () => task.stop());
  fastify.log.info('Price scheduler enabled (*/15 * * * *)');
});
```

```ts
import Decimal from 'decimal.js';
import { and, desc, eq, gte, lte, count } from 'drizzle-orm';

import { db } from '../../db/index.js';
import { goldTransactions, priceCache, type GoldTransaction } from '../../db/schema/index.js';
import { computeHolding, pricePerChi, toChi, unrealizedPnl, type GoldTx } from './goldMath.js';
import type { CreateGoldTxBody, ListGoldTxQuery, UpdateGoldTxBody } from './gold.schema.js';

const STALE_MS = 20 * 60 * 1000;

export class GoldError extends Error {
  constructor(
    public readonly statusCode: number,
    message: string,
  ) {
    super(message);
  }
}

function toEngineTx(t: GoldTransaction): GoldTx {
  return {
    action: t.action,
    quantity: t.quantity,
    unit: t.unit,
    pricePerUnit: t.pricePerUnit,
    fee: t.fee,
    transactionAt: t.transactionAt,
  };
}

export const goldService = {
  async list(userId: string, q: ListGoldTxQuery) {
    const conds = [eq(goldTransactions.userId, userId)];
    if (q.goldType) conds.push(eq(goldTransactions.goldType, q.goldType));
    if (q.action) conds.push(eq(goldTransactions.action, q.action));
    if (q.from) conds.push(gte(goldTransactions.transactionAt, q.from));
    if (q.to) conds.push(lte(goldTransactions.transactionAt, q.to));
    const where = and(...conds);

    const [rows, [{ value: total }]] = await Promise.all([
      db
        .select()
        .from(goldTransactions)
        .where(where)
        .orderBy(desc(goldTransactions.transactionAt))
        .limit(q.limit)
        .offset((q.page - 1) * q.limit),
      db.select({ value: count() }).from(goldTransactions).where(where),
    ]);
    return { data: rows, pagination: { page: q.page, limit: q.limit, total: Number(total) } };
  },

  async create(userId: string, body: CreateGoldTxBody): Promise<GoldTransaction> {
    if (body.action === 'sell') {
      const existing = await db
        .select()
        .from(goldTransactions)
        .where(and(eq(goldTransactions.userId, userId), eq(goldTransactions.goldType, body.goldType)));
      const held = computeHolding(existing.map(toEngineTx)).heldChi;
      const sellChi = toChi(body.quantity, body.unit);
      if (sellChi.gt(held)) {
        throw new GoldError(400, 'Sell exceeds current holdings');
      }
    }
    const [row] = await db
      .insert(goldTransactions)
      .values({
        userId,
        goldType: body.goldType,
        action: body.action,
        quantity: String(body.quantity),
        unit: body.unit,
        pricePerUnit: String(body.pricePerUnit),
        fee: String(body.fee),
        storage: body.storage,
        note: body.note,
        ...(body.transactionAt ? { transactionAt: body.transactionAt } : {}),
      })
      .returning();
    return row!;
  },

  async update(userId: string, id: string, patch: UpdateGoldTxBody): Promise<GoldTransaction> {
    const set: Record<string, unknown> = {};
    if (patch.goldType !== undefined) set.goldType = patch.goldType;
    if (patch.action !== undefined) set.action = patch.action;
    if (patch.quantity !== undefined) set.quantity = String(patch.quantity);
    if (patch.unit !== undefined) set.unit = patch.unit;
    if (patch.pricePerUnit !== undefined) set.pricePerUnit = String(patch.pricePerUnit);
    if (patch.fee !== undefined) set.fee = String(patch.fee);
    if (patch.storage !== undefined) set.storage = patch.storage;
    if (patch.note !== undefined) set.note = patch.note;
    if (patch.transactionAt !== undefined) set.transactionAt = patch.transactionAt;

    const [row] = await db
      .update(goldTransactions)
      .set(set)
      .where(and(eq(goldTransactions.id, id), eq(goldTransactions.userId, userId)))
      .returning();
    if (!row) throw new GoldError(404, 'Transaction not found');
    return row;
  },

  async remove(userId: string, id: string): Promise<void> {
    const [row] = await db
      .delete(goldTransactions)
      .where(and(eq(goldTransactions.id, id), eq(goldTransactions.userId, userId)))
      .returning();
    if (!row) throw new GoldError(404, 'Transaction not found');
  },

  async portfolio(userId: string) {
    const rows = await db
      .select()
      .from(goldTransactions)
      .where(eq(goldTransactions.userId, userId));

    const prices = await db.select().from(priceCache).where(eq(priceCache.assetType, 'gold'));
    const priceBySymbol = new Map(prices.map((p) => [p.symbol, p]));

    const byType = new Map<string, GoldTransaction[]>();
    for (const r of rows) {
      const list = byType.get(r.goldType) ?? [];
      list.push(r);
      byType.set(r.goldType, list);
    }

    type Row = {
      goldType: string;
      heldChi: string;
      dcaPerChi: string;
      currentPricePerChi: string | null;
      value: string | null;
      pnl: string | null;
      pnlPct: string | null;
      weightPct: string | null;
    };
    const holdings: Row[] = [];
    let totalValue = new Decimal(0);
    let totalInvested = new Decimal(0);

    for (const [goldType, txs] of byType) {
      const h = computeHolding(txs.map(toEngineTx));
      if (h.heldChi.isZero()) continue;
      const priceRow = priceBySymbol.get(goldType);
      const current = priceRow?.priceBuy ? new Decimal(priceRow.priceBuy) : null;
      const value = current ? current.mul(h.heldChi) : null;
      const pnlObj = current ? unrealizedPnl(h.heldChi, h.dcaPerChi, current) : null;
      totalInvested = totalInvested.plus(h.investedRemaining);
      if (value) totalValue = totalValue.plus(value);
      holdings.push({
        goldType,
        heldChi: h.heldChi.toString(),
        dcaPerChi: h.dcaPerChi.toFixed(2),
        currentPricePerChi: current ? current.toFixed(2) : null,
        value: value ? value.toFixed(2) : null,
        pnl: pnlObj ? pnlObj.pnl.toFixed(2) : null,
        pnlPct: pnlObj ? pnlObj.pnlPct.toFixed(2) : null,
        weightPct: null, // filled below
      });
    }

    for (const row of holdings) {
      row.weightPct =
        row.value && !totalValue.isZero()
          ? new Decimal(row.value).div(totalValue).mul(100).toFixed(2)
          : null;
    }

    const totalPnl = totalValue.minus(totalInvested);
    const totals = {
      value: totalValue.toFixed(2),
      invested: totalInvested.toFixed(2),
      pnl: totalPnl.toFixed(2),
      pnlPct: totalInvested.isZero() ? '0.00' : totalPnl.div(totalInvested).mul(100).toFixed(2),
    };
    return { holdings, totals };
  },

  async prices() {
    const rows = await db.select().from(priceCache).where(eq(priceCache.assetType, 'gold'));
    const updatedAt = rows.reduce<Date | null>(
      (max, r) => (!max || r.fetchedAt > max ? r.fetchedAt : max),
      null,
    );
    const stale = !updatedAt || Date.now() - updatedAt.getTime() > STALE_MS;
    return {
      prices: rows.map((r) => ({
        symbol: r.symbol,
        source: r.source,
        priceBuy: r.priceBuy,
        priceSell: r.priceSell,
        currency: r.currency,
        fetchedAt: r.fetchedAt,
      })),
      updatedAt,
      stale,
    };
  },
};
```

```tsx
                <option value="chi">Chỉ</option>
                <option value="luong">Lượng</option>
                <option value="cay">Cây</option>
              </select>
            </div>
            <p className="mt-1 text-xs text-neutral-500">= {qtyChi} Chỉ</p>
          </div>

          <div>
            <label className="mb-1 block text-sm text-neutral-400">Giá giao dịch (VND/{unit})</label>
            <input type="number" min="0" value={pricePerUnit} onChange={(e) => setPricePerUnit(e.target.value)} required className={`${input} font-mono`} />
          </div>

          <div>
            <label className="mb-1 block text-sm text-neutral-400">Phí (tùy chọn)</label>
            <input type="number" min="0" value={fee} onChange={(e) => setFee(e.target.value)} className={`${input} font-mono`} />
          </div>

          <div>
            <label className="mb-1 block text-sm text-neutral-400">Nơi lưu trữ</label>
            <select value={storage} onChange={(e) => setStorage(e.target.value)} className={input}>
              {STORAGES.map((s) => (
                <option key={s} value={s}>{s}</option>
              ))}
            </select>
          </div>
        </div>

        <label className="mb-1 mt-4 block text-sm text-neutral-400">Ghi chú</label>
        <textarea value={note} maxLength={500} onChange={(e) => setNote(e.target.value)} className={input} rows={2} />

        <div className="mt-6 flex gap-3">
          <button type="button" onClick={() => navigate({ to: '/gold' })} className="rounded-md border border-neutral-700 px-5 py-2 text-sm text-neutral-300 hover:bg-neutral-800">
            Huỷ
          </button>
          <button type="submit" disabled={saving} className="rounded-md bg-brand px-5 py-2 text-sm font-medium text-white hover:bg-brand-dark disabled:opacity-50">
            {saving ? 'Đang lưu...' : 'Lưu giao dịch'}
          </button>
        </div>
      </form>
    </div>
  );
}
```
